# ui-audit v2 — Hierarchical YAML Fingerprints with Accessibility Tree

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Upgrade ui-audit from a flat JSON property extractor to a hierarchical YAML fingerprint tool that combines accessibility tree structure with computed visual properties, enabling cross-version UI comparison for AI agents.

**Architecture:** Playwright crawls the running app, extracts the Chrome accessibility tree via CDP, augments each node with computed visual properties (colors, dimensions, overflow), groups into semantic regions (ARIA landmarks), and serializes to a hierarchical YAML format. Three commands: `capture` (generate fingerprint), `verify` (diff against baseline), `diff` (compare two fingerprints). Output includes per-component cropped PNGs for agent visual inspection.

**Tech Stack:** TypeScript, Playwright, Chrome DevTools Protocol (CDP), `yaml` npm package, Node.js CLI (`commander` or minimal arg parsing)

```typescript
// src/extract/accessibility.ts
import type { Page } from 'playwright'

export interface AXNode {
  role: string
  name: string
  value?: string
  description?: string
  children?: AXNode[]
  backendDOMNodeId?: number
  properties?: Record<string, string | boolean | number>
}

/**
 * Extract the full accessibility tree from a Playwright page via CDP.
 *
 * Uses Chrome DevTools Protocol Accessibility.getFullAXTree which returns
 * every node in the accessibility tree with roles, names, and properties.
 */
export async function extractAccessibilityTree(page: Page): Promise<AXNode> {
  const client = await page.context().newCDPSession(page)

  const { nodes } = await client.send('Accessibility.getFullAXTree')

  // Build a tree from the flat CDP node list
  const nodeMap = new Map<string, AXNode & { parentId?: string }>()

  for (const node of nodes) {
    const axNode: AXNode & { parentId?: string } = {
      role: node.role?.value ?? 'none',
      name: node.name?.value ?? '',
      value: node.value?.value,
      description: node.description?.value,
      children: [],
      backendDOMNodeId: node.backendDOMNodeId,
      properties: {},
    }

    // Extract boolean/string properties (checked, disabled, expanded, etc.)
    if (node.properties) {
      for (const prop of node.properties) {
        axNode.properties![prop.name] = prop.value?.value
      }
    }

    if (node.childIds?.length) {
      // Will connect children in second pass
    }

    nodeMap.set(node.nodeId, axNode)

    // Track parent relationship
    if (node.parentId) {
      axNode.parentId = node.parentId
    }
  }

  // Connect children
  for (const [id, node] of nodeMap) {
    if (node.parentId && nodeMap.has(node.parentId)) {
      const parent = nodeMap.get(node.parentId)!
      parent.children!.push(node)
    }
  }

  // Find root (node with no parent)
  for (const node of nodeMap.values()) {
    if (!node.parentId) {
      return node
    }
  }

  // Fallback: return first node
  return nodeMap.values().next().value!
}

/**
 * Extract only ARIA landmark regions from the tree.
 * Returns a flat list of nodes with landmark roles.
 */
const LANDMARK_ROLES = new Set([
  'banner', 'navigation', 'main', 'complementary', 'contentinfo', 'form', 'search',
])

export function extractLandmarks(tree: AXNode): AXNode[] {
  const landmarks: AXNode[] = []
  function walk(node: AXNode) {
    if (LANDMARK_ROLES.has(node.role)) landmarks.push(node)
    for (const child of node.children ?? []) walk(child)
  }
  walk(tree)
  return landmarks
}
```

```typescript
// src/extract/visual-properties.ts
import type { Page } from 'playwright'
import type { Bounds } from '../fingerprint/types'

export interface VisualProps {
  bounds: Bounds
  visible: boolean
  backgroundColor: string
  color: string
  fontSize: string
  borderWidth: string
  opacity: string
  display: string
  overflow: string
  textOverflow: boolean
  textContent: string
  childCount: number
}

/**
 * Extract computed visual properties for a single DOM element.
 * Returns null if the element doesn't exist.
 */
export async function extractVisualProperties(
  page: Page,
  selector: string,
): Promise<VisualProps | null> {
  const locator = page.locator(selector).first()
  const count = await page.locator(selector).count()
  if (count === 0) return null

  return locator.evaluate((el) => {
    const rect = el.getBoundingClientRect()
    const cs = window.getComputedStyle(el)
    const h = el as HTMLElement
    return {
      bounds: {
        x: Math.round(rect.x),
        y: Math.round(rect.y),
        width: Math.round(rect.width),
        height: Math.round(rect.height),
      },
      visible:
        rect.width > 0 &&
        rect.height > 0 &&
        cs.visibility !== 'hidden' &&
        cs.display !== 'none',
      backgroundColor: cs.backgroundColor,
      color: cs.color,
      fontSize: cs.fontSize,
      borderWidth: cs.borderWidth,
      opacity: cs.opacity,
      display: cs.display,
      overflow: cs.overflow,
      textOverflow: h.scrollWidth > h.clientWidth,
      textContent: (h.textContent ?? '').trim().substring(0, 200),
      childCount: h.children.length,
    }
  })
}

/**
 * Extract visual properties for multiple selectors.
 * Returns a map of selector → VisualProps.
 */
export async function extractAllVisualProperties(
  page: Page,
  selectors: string[],
): Promise<Map<string, VisualProps>> {
  const results = new Map<string, VisualProps>()
  for (const selector of selectors) {
    const props = await extractVisualProperties(page, selector)
    if (props) results.set(selector, props)
  }
  return results
}
```

```typescript
// src/extract/merge.ts
import type { Page } from 'playwright'
import type { UIFingerprint, Region, Component, ElementProps } from '../fingerprint/types'
import { extractAccessibilityTree, extractLandmarks, type AXNode } from './accessibility'
import { extractVisualProperties } from './visual-properties'

/**
 * Build a complete UIFingerprint by merging the accessibility tree
 * with computed visual properties.
 */
export async function buildFingerprint(page: Page): Promise<UIFingerprint> {
  const url = page.url()
  const title = await page.title()
  const viewport = page.viewportSize() ?? { width: 1440, height: 900 }

  // Extract accessibility tree
  const axTree = await extractAccessibilityTree(page)
  const landmarks = extractLandmarks(axTree)

  // Extract page-level visual properties
  const bodyProps = await extractVisualProperties(page, 'body')
  const background = bodyProps?.backgroundColor ?? ''

  // Build regions from landmarks
  const regions: Record<string, Region> = {}

  for (const landmark of landmarks) {
    const selector = landmarkToSelector(landmark)
    const props = await extractVisualProperties(page, selector)
    if (!props) continue

    // Count direct children in the landmark
    const childCount = await page.locator(selector).first().evaluate(
      el => el.children.length
    )

    // Build component list from landmark's children in the accessibility tree
    const components: Component[] = []
    for (const child of landmark.children ?? []) {
      if (child.role === 'none' || child.role === 'generic') continue
      components.push({
        id: child.name || child.role,
        selector: '',
        props: {
          role: child.role,
          name: child.name,
          bounds: props.bounds, // approximation — full version resolves per-child
          visible: true,
          backgroundColor: '',
          color: '',
          fontSize: '',
          borderWidth: '',
          opacity: '1',
          display: '',
          overflow: '',
          textOverflow: false,
          textContent: child.name,
          childCount: child.children?.length ?? 0,
        },
      })
    }

    const regionKey = landmark.role + (landmark.name ? `-${slugify(landmark.name)}` : '')
    regions[regionKey] = {
      role: landmark.role,
      bounds: props.bounds,
      background: props.backgroundColor,
      childCount,
      summary: '', // populated in a later task
      components,
    }
  }

  return {
    version: 2,
    page: {
      url,
      title,
      viewport,
      theme: '', // populated by theme detection in a later task
      background,
      layout: inferLayout(regions),
      landmarks: landmarks.map(l => l.role),
      capturedAt: new Date().toISOString(),
    },
    regions,
    ungrouped: [],
    state: { name: 'default', modals: 'none', selection: null },
  }
}

function landmarkToSelector(node: AXNode): string {
  // Map ARIA roles to typical HTML selectors
  const roleMap: Record<string, string> = {
    banner: 'header, [role="banner"]',
    navigation: 'nav, [role="navigation"]',
    main: 'main, [role="main"]',
    complementary: 'aside, [role="complementary"]',
    contentinfo: 'footer, [role="contentinfo"]',
    search: '[role="search"]',
    form: 'form, [role="form"]',
  }
  return roleMap[node.role] ?? `[role="${node.role}"]`
}

function inferLayout(regions: Record<string, Region>): string {
  const parts: string[] = []
  if (Object.keys(regions).some(k => k.startsWith('banner'))) parts.push('header')
  if (Object.keys(regions).some(k => k.startsWith('navigation'))) parts.push('sidebar')
  if (Object.keys(regions).some(k => k.startsWith('main'))) parts.push('main')
  if (Object.keys(regions).some(k => k.startsWith('contentinfo'))) parts.push('footer')
  return parts.join(' + ')
}

function slugify(s: string): string {
  return s.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-|-$/g, '').substring(0, 30)
}
```

```typescript
// src/fingerprint/diff-engine.ts
import type { UIFingerprint, DiffReport, DiffFailure, Component } from './types'
import { EXACT_COMPARE_PROPS, NUMERIC_TOLERANCES, INVARIANTS } from '../constants'

/**
 * Compare two fingerprints and produce a structured diff report.
 * Runs invariant checks on `current`, then diffs against `baseline`.
 */
export function diffFingerprints(
  baseline: UIFingerprint,
  current: UIFingerprint,
): DiffReport {
  const failures: DiffFailure[] = []
  const baseRegions = Object.keys(baseline.regions)
  const currRegions = Object.keys(current.regions)
  const missing = baseRegions.filter(k => !currRegions.includes(k))
  const added = currRegions.filter(k => !baseRegions.includes(k))

  let totalChecked = 0

  // Check invariants on all current components
  for (const [regionKey, region] of Object.entries(current.regions)) {
    for (const comp of region.components) {
      for (const inv of INVARIANTS) {
        totalChecked++
        const value = comp.props[inv.prop as keyof typeof comp.props]
        if (inv.check(value, comp.props.role)) {
          failures.push({
            component: comp.id,
            region: regionKey,
            property: inv.prop,
            expected: `not ${String(value)}`,
            actual: value as string | number | boolean,
            screenshotFile: comp.props.screenshotFile,
          })
        }
      }
    }
  }

  // Diff components against baseline
  for (const regionKey of baseRegions) {
    if (!current.regions[regionKey]) continue
    const baseComps = baseline.regions[regionKey].components
    const currComps = current.regions[regionKey].components

    for (const baseComp of baseComps) {
      const currComp = currComps.find(c => c.id === baseComp.id)
      if (!currComp) {
        failures.push({
          component: baseComp.id,
          region: regionKey,
          property: 'exists',
          expected: true,
          actual: false,
        })
        totalChecked++
        continue
      }

      // Exact property comparison
      for (const prop of EXACT_COMPARE_PROPS) {
        totalChecked++
        const bVal = baseComp.props[prop as keyof typeof baseComp.props]
        const cVal = currComp.props[prop as keyof typeof currComp.props]
        if (bVal !== cVal) {
          failures.push({
            component: currComp.id,
            region: regionKey,
            property: prop,
            expected: bVal as string | number | boolean,
            actual: cVal as string | number | boolean,
            screenshotFile: currComp.props.screenshotFile,
          })
        }
      }

      // Numeric tolerance comparison
      for (const [prop, tolerance] of Object.entries(NUMERIC_TOLERANCES)) {
        totalChecked++
        const bVal = (baseComp.props.bounds as Record<string, number>)[prop]
        const cVal = (currComp.props.bounds as Record<string, number>)[prop]
        if (typeof bVal === 'number' && typeof cVal === 'number') {
          if (Math.abs(bVal - cVal) > tolerance) {
            failures.push({
              component: currComp.id,
              region: regionKey,
              property: `bounds.${prop}`,
              expected: bVal,
              actual: cVal,
              screenshotFile: currComp.props.screenshotFile,
            })
          }
        }
      }
    }
  }

  // Missing regions
  for (const key of missing) {
    totalChecked++
    failures.push({
      component: key,
      region: key,
      property: 'region.exists',
      expected: true,
      actual: false,
    })
  }

  return {
    pass: failures.length === 0,
    timestamp: new Date().toISOString(),
    old: baseline.page.url,
    new: current.page.url,
    totalChecked,
    passed: totalChecked - failures.length,
    failed: failures.length,
    missing,
    added,
    failures,
  }
}
```

```typescript
// src/extract/screenshots.ts
import type { Page } from 'playwright'
import { join } from 'path'
import { mkdirSync } from 'fs'

/**
 * Capture a cropped PNG screenshot of a single DOM element.
 * Returns the relative filename, or null if the element doesn't exist or is invisible.
 */
export async function captureComponentScreenshot(
  page: Page,
  selector: string,
  outDir: string,
  slug: string,
): Promise<string | null> {
  const locator = page.locator(selector).first()
  if (await page.locator(selector).count() === 0) return null
  if (!(await locator.isVisible())) return null

  mkdirSync(outDir, { recursive: true })
  const filename = `${slug}.png`
  const filepath = join(outDir, filename)

  try {
    await locator.screenshot({
      path: filepath,
      type: 'png',
      animations: 'disabled',
    })
    return filename
  } catch {
    return null
  }
}

/**
 * Capture a full-page screenshot for reference.
 */
export async function captureFullPage(
  page: Page,
  outDir: string,
): Promise<string> {
  mkdirSync(outDir, { recursive: true })
  const filepath = join(outDir, 'full-page.png')
  await page.screenshot({ path: filepath, fullPage: true, type: 'png' })
  return 'full-page.png'
}
```

- [ ] **Step 2: Create capture.ts**

```typescript
// src/capture.ts — orchestrates: launch browser → buildFingerprint → capture screenshots → serialize YAML → write files
export async function runCapture(url: string, outDir: string): Promise<void>
```

Steps inside:
1. Launch Chromium with viewport 1440x900, dark color scheme, animations disabled
2. Navigate to URL, wait for networkidle
3. Call `buildFingerprint(page)` to get the UIFingerprint
4. For each region's components, call `captureComponentScreenshot()` and attach the filename to the component
5. Call `captureFullPage()` for reference
6. Serialize fingerprint to YAML via `serializeFingerprint()`
7. Write `fingerprint.yaml` + screenshots to outDir
8. Close browser, print summary

- [ ] **Step 3: Create verify.ts**

```typescript
// src/verify.ts — orchestrates: capture current → load baseline → diff → write report
export async function runVerify(url: string, baselineDir: string): Promise<boolean>
```

Steps inside:
1. Load baseline from `{baselineDir}/fingerprint.yaml` via `deserializeFingerprint()`
2. Capture current state (same as capture.ts)
3. Call `diffFingerprints(baseline, current)` to get report
4. Write report to `{baselineDir}/report.json`
5. Print pass/fail summary with per-failure details
6. Return boolean (pass/fail)

```typescript
// src/diff.ts — compare two fingerprints without a running app
export async function runDiff(oldDir: string, newDir: string): Promise<void>
```

Steps inside:

- [ ] **Step 6: Write integration test**

```typescript
// tests/cli.test.ts
import { describe, it, expect } from 'vitest'
import { execSync } from 'child_process'
import { existsSync, readFileSync, rmSync } from 'fs'
import { join } from 'path'

const OUT_DIR = join(__dirname, '../.test-audit')
const FIXTURE = `file://${join(__dirname, '../fixtures/sample-page.html')}`

describe('CLI integration', () => {
  it('capture produces fingerprint.yaml + screenshots', () => {
    if (existsSync(OUT_DIR)) rmSync(OUT_DIR, { recursive: true })
    execSync(`npx tsx src/cli.ts capture --url "${FIXTURE}" --out ${OUT_DIR}`, {
      cwd: join(__dirname, '..'),
    })
    expect(existsSync(join(OUT_DIR, 'fingerprint.yaml'))).toBe(true)
    expect(existsSync(join(OUT_DIR, 'screenshots'))).toBe(true)

    const yaml = readFileSync(join(OUT_DIR, 'fingerprint.yaml'), 'utf-8')
    expect(yaml).toContain('version: 2')
    expect(yaml).toContain('role: banner')
  })

  it('verify against own baseline passes (minus known invariant failures)', () => {
    // Verify against the baseline we just captured
    try {
      execSync(`npx tsx src/cli.ts verify --url "${FIXTURE}" --baseline ${OUT_DIR}`, {
        cwd: join(__dirname, '..'),
      })
    } catch {
      // May fail on invariant checks — that's OK for this test
    }
    expect(existsSync(join(OUT_DIR, 'report.json'))).toBe(true)
  })
})
```
